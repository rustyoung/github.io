golang排序示例

package main

import (
    "fmt"
    "sort"
)

type SortKey struct {
    Key  string
    Data []map[string]string
}

type lessFunc func(p1, p2 *map[string]interface{}) bool

type SortKeys struct {
    Data []map[string]interface{}
    less []lessFunc
}

func main() {
    a := []int{0, 9, 1, 8, -1, 30, 22}
    sort.Sort(sort.IntSlice(a))
    fmt.Println(a)

    a1 := []int{0, 9, 1, 8, -1, 30, 22}
    sort.Sort(sort.Reverse(sort.IntSlice(a1)))
    fmt.Println(a1)

    b := []int{0, 9, 1, 8, -1, 30, 22}
    sort.Ints(b)
    fmt.Println(b)

    c := []string{"a", "1b", "c", "c123", "c1", "e"}
    sort.Strings(c)
    fmt.Println(c)

    d := []string{"a", "1b", "c", "c123", "c1", "e"}
    sort.Sort(sort.StringSlice(d))
    fmt.Println(d)

    d1 := []string{"a", "1b", "c", "c123", "c1", "e"}
    sort.Sort(sort.Reverse(sort.StringSlice(d1)))
    fmt.Println(d1)

    data := make([]map[string]string, 0)
    data = append(data, map[string]string{"id": "9", "name": "lucy"})
    data = append(data, map[string]string{"id": "1", "name": "lily"})
    data = append(data, map[string]string{"id": "6", "name": "nancy"})
    data = append(data, map[string]string{"id": "3", "name": "daisy"})
    data = append(data, map[string]string{"id": "7", "name": "tiffany"})
    fmt.Println(sortKey(data, "id"))

    ss := []map[string]interface{}{
            map[string]interface{}{
                "id":    1,
                "money": 100,
                "name":  "jim",
            },
            map[string]interface{}{
                "id":    2,
                "money": 100,
                "name":  "ann",
            },
            map[string]interface{}{
                "id":    3,
                "money": 80,
                "name":  "ab",
            },
            map[string]interface{}{
                "id":    4,
                "money": 1000,
                "name":  "rust",
            },
         }

    idSort := func(c1, c2 *map[string]interface{}) bool {
        return (*c1)["id"].(int) < (*c2)["id"].(int)
    }
    moneySort := func(c1, c2 *map[string]interface{}) bool {
        return (*c1)["money"].(int) < (*c2)["money"].(int)
    }
    nameSort := func(c1, c2 *map[string]interface{}) bool {
        return (*c1)["name"].(string) < (*c2)["name"].(string)
    }

    orderBy(nameSort, moneySort).Sort(ss)
    fmt.Println(ss)

    orderBy(moneySort, idSort).Sort(ss)
    fmt.Println(ss)
}

func sortKey(data []map[string]string, key string) []map[string]string {
    d := &SortKey{key, data}
    sort.Sort(d)
    return d.Data
}

func (s *SortKey) Len() int {
    return len(s.Data)  
}

func (s *SortKey) Swap(i int, j int) {
    s.Data[i], s.Data[j] = s.Data[j], s.Data[i]
}

func (s *SortKey) Less(i int, j int) bool {
    return s.Data[i][s.Key] < s.Data[j][s.Key]
}

func (s *SortKeys) Sort(data []map[string]interface{}) {
    s.Data = data
    sort.Sort(s)
}

func orderBy(less ...lessFunc) *SortKeys {
    return &SortKeys{less: less}
}

func (s *SortKeys) Len() int {
    return len(s.Data)
}

func (s *SortKeys) Swap(i, j int) {
    s.Data[i], s.Data[j] = s.Data[j], s.Data[i]
}

func (s *SortKeys) Less(i, j int) bool {
    p, q := &s.Data[i], &s.Data[j]
    var k int
    for k = 0; k < len(s.less)-1; k++ {
        less := s.less[k]
        switch {
        case less(p, q):
            return true
        case less(q, p):
            return false
        }
    }
    return s.less[k](p, q)
}
